document.addEventListener('DOMContentLoaded', function() {
    
    const adminForm = document.getElementById('adminLogin');
    
    adminForm.addEventListener('submit', function(e) {
        e.preventDefault();
        
        const username = document.getElementById('username').value;
        const password = document.getElementById('password').value;
        const remember = document.querySelector('.remember-forgot input[type="checkbox"]').checked;
        
        // Validasi
        if (!username || !password) {
            alert('Mohon isi username dan password!');
            return;
        }
        
        // Cek kredensial
        if (username === 'admin' && password === 'admin123') {
            alert('Login berhasil! Selamat datang Administrator.');
            console.log('Admin login:', { username, remember });
            
            // Redirect ke dashboard admin
            setTimeout(() => {
                window.location.href = '/dashboard-admin.html';
            }, 1500);
        } 
        else if (username === 'petugas' && password === 'petugas123') {
            alert('Login berhasil! Selamat datang Petugas.');
            console.log('Petugas login:', { username, remember });
            
            // Redirect ke dashboard petugas
            setTimeout(() => {
                window.location.href = '/dashboard-petugas.html';
            }, 1500);
        }
        else {
            alert('Username atau password salah!');
        }
    });
    
    // ============ TOGGLE PASSWORD ============
    const togglePassword = document.querySelector('.toggle-password');
    
    if (togglePassword) {
        togglePassword.addEventListener('click', function() {
            const passwordInput = document.getElementById('password');
            const type = passwordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            passwordInput.setAttribute('type', type);
            
            // Toggle icon
            this.classList.toggle('fa-eye');
            this.classList.toggle('fa-eye-slash');
        });
    }
    
    // ============ FORGOT PASSWORD ============
    const forgotLink = document.getElementById('forgotPassword');
    
    if (forgotLink) {
        forgotLink.addEventListener('click', function(e) {
            e.preventDefault();
            alert('Silakan hubungi Super Administrator:\n\nEmail: superadmin@perpustakaan.id\nTelp: (021) 555-1234');
        });
    }
    
});
